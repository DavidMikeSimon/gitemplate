# Custom variables via `--json`

    gitemplate --name my-new-project \
               --src git@github.com:me/my-old-template.git \
               --dst ~/dev/my-new-project \
               --desc 'gets it done' \
               --json '{"customVar1":"val1","customVar2":"val2"}' \
